<script>
		let divNode = null;
		let animationCounter = 0;
		let closed = false;

		window.addEventListener("load", () => {
			divNode = document.querySelector(".pop-up");

			let date = new Date();
			console.log(date.getTime());

			divNode.onsubmit = () => {
				divNode.style.display = "none";
			} 

			tick();
		});

		const tick = () => {
			animationCounter++;
			// offsetTop, offsetLeft, offsetWidth, offsetHeight...
			let pos = divNode.offsetTop + 1;
			divNode.style.top = (pos > 100 ? 100 : pos) + "px";
			
			if (pos > 100 && !closed )
			{
			divNode.style.opacity= 0.9;
			}

			window.requestAnimationFrame(tick); // 60fps
		}
		
		function closepopup(){
		
			divNode.style.display="none";
			closed = true;
			console.log("bonheur et alégresse");
		
		}
		
	</script>
